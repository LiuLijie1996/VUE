### 时间格式转换成：年月日、年月

```js
// 中国标准时间format yyyy-mm-dd
const format = (time) => {
    let ymd = '';

    //得到传进来的时间的月份
    let mouth = (time.getMonth() + 1) >= 10 ? (time.getMonth() + 1) : ('0' + (time.getMonth() + 1));
    //得到传进来的时间的日
    let day = time.getDate() >= 10 ? time.getDate() : ('0' + time.getDate());
    //得到传进来的时间的年份，然后拼接 月-日
    ymd += time.getFullYear() + '-'; // 获取年份。
    ymd += mouth + '-'; // 获取月份。
    ymd += day; // 获取日。
    return ymd // 返回日期。
};

//获取开始与结束中间的 年月日（yyyy-mm-dd）
const getAllDate = (start, end) => {
    let dateArr = [];
    let startArr = start.split('-');
    let endArr = end.split('-');

    let db = new Date(startArr[0], startArr[1] - 1, startArr[2]);//得到开始时间
    let de = new Date(endArr[0], endArr[1] - 1, endArr[2]);//得到结束时间

    let unixDb = db.getTime();//开始时间戳
    let unixDe = de.getTime();//结束时间戳

    const oneDay = 24 * 60 * 60 * 1000;//一天的时间戳
    //定义开始时间=开始时间戳；开始时间<=结束时间戳；将开始时间增加一天的时间戳
    for (let stamp = unixDb; stamp <= unixDe; stamp += oneDay) {
        //调用format函数将时间传入进去，返回：年-月-日
        dateArr.push(format(new Date(parseInt(stamp))));
    }
    return dateArr
};
//调用getAllDate获取开始时间到结束时间的所有 年月日
console.log(getAllDate('2018-12-12', '2019-03-03'));


//切割年月日只保留：年月（yyyy-mm）
function getAllMonth(getAllDate) {
    let dateArr = [];

    let reg = /(.*)-\d/;//用来过滤 年月的
    getAllDate.forEach(item => {
        //过滤出年月：yyyy-mm
        item.replace(reg, function (item, y_m) {
            //判断是否有重复项，没有就添加日期
            if (dateArr[dateArr.length ? dateArr.length - 1 : 0] !== y_m) {
                dateArr.push(y_m);
            }
        })
    });

    return dateArr;
}
```

### 安装

```
npm install vue-lazyload --save
```



### main.js中引入

```js
//图片懒加载
import VueLazyLoad from 'vue-lazyload';
Vue.use(VueLazyLoad, {
    //设置占位图片的文件地址
    loading: require("assets/img/common/placeholder.png"),
});
```



### 组件中使用

```vue
<!-- 将src换成 v-lazy -->
<img v-lazy="item.src" alt="">
```
